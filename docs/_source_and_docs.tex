\hypertarget{usage-and-examples-1}{%
\subsection{Usage and examples}\label{usage-and-examples-1}}

\begin{minted}[]{matlab}
sys = tf([5,3,5],[1,6,1,20])
\end{minted}

\begin{minted}[]{text}
sys =
 
    5 s^2 + 3 s + 5
  --------------------
  s^3 + 6 s^2 + s + 20
 
Continuous-time transfer function.
\end{minted}

\begin{minted}[]{matlab}
sys_a = tf_factor(sys)
\end{minted}

\begin{minted}[]{text}
sys_a(:,:,1,1) =
 
          3.155
  ----------------------
  s^2 - 0.3399 s + 3.155
 

sys_a(:,:,2,1) =
 
    6.34
  --------
  s + 6.34
 

sys_a(:,:,3,1) =
 
  s^2 + 0.6 s + 1
 

sys_a(:,:,4,1) =
 
  0.25
 
4x1 array of continuous-time transfer functions.
\end{minted}

\begin{minted}[]{matlab}
% [f,ax_mag,ax_phase] = bode_multi(G); % get axis handles
f = bode_multi(sys_a);

hgsave(f,'figures/temp');
\end{minted}
